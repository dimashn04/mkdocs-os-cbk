```bash
WEEK=Week02
DEMOS=$HOME/git/demos/Demos/
if [ -d $DEMOS ] ; then
  cd $DEMOS
  git pull
  mkdir -vp $HOME/mydemo/
  cd $HOME/mydemo/
  cp -r $DEMOS/$WEEK/ .
  cd $WEEK/
  ls -al
else
  echo "ERROR: no $DEMOS directory!"
fi
```
